def base_uri(params, options)
  path = "#{api}/clu_patient_document"
  path << '/debug/true' if options[:debug]
  if params[:id]
    path << "/#{params[:id]}"
  elsif params[:patient_document_id]
    path << "/patient_document_id/#{params[:patient_document_id]}"
  else
    raise Exceptions::BadRequestError, 'Must provide an id or a patient_document_id'
  end
  path
end

def data(params, options={})
  path = "#{base_uri(params, options)}/data"
  request(path, options).if_404_raise(Exceptions::PatientDocumentNotFoundError)
                        .to_s
end
