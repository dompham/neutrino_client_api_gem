        def build
raise Exceptions::BadRequestError if @specified_whitelists.count > 1
whitelist = @specified_whitelists.first
values = @whitelist_hash[whitelist]

case whitelist

when :type_of_service_whitelist
  WhitelistUri.new
           .with_template('/type_of_service_list/{value}')
           .and_values(values)
           .error_on_empty(Exceptions::TypesOfServiceNotProvided)

when :subject_matter_domain_whitelist
  WhitelistUri.new
           .with_template('/subject_matter_domain_list/{value}')
           .and_values(values)
           .error_on_empty(Exceptions::SubjectMatterDomainsNotProvided)

when :with_ejection_fractions
  WhitelistUri.new.with_template('/with/ejection_fractions')

when :snomed_problem_whitelist
  WhitelistUri.new
           .with_template("/problem_whitelist/snomed/{value}")
           .and_value(values)
           .append_component(clinical_or_all)
           .error_on_empty(Exceptions::SnomedCodesNotProvided)

when :snomed_procedure_whitelist
